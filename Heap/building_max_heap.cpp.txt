void heapify(int arr[],int size,int i){
    int largest=i;
    int left=2*i;
    int right=2*i+1;

    if(left < size && arr[largest]<arr[left]){
        largest=left;
    }
    if(right < size && arr[largest]<arr[right]){
        largest=right;
    }

    if(largest!=i){
        swap(arr[largest],arr[i]);
        heapify(arr,size,largest);
    }
}
