class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(head==NULL){
            return head;
        }
        ListNode* temp=head;
        while(temp->next!=nullptr ){
            if(temp->val==temp->next->val){
                 // while(temp->val==temp->next->val){
               temp->next=temp->next->next;
            }
            // }
           else{
               temp=temp->next;
           }
            
        }
        return head;
    }
};
